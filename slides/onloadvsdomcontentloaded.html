onload vs DOMContentLoaded

기다리는 이벤트에는 onload와 DOMContentLoaded 두가지 방법이 있다.

 
window.onload = init;
//or
//window.addEventListener('load', init)
function init () {
  var wowElement = document.getElementById('wow');
  wow.addEventListener('click', function () {
    alert("wow");
  })
}

 
window.addEventListener('DOMContentLoaded', init)
function init () {
  var wowElement = document.getElementById('wow');
  wow.addEventListener('click', function () {
    alert("wow");
  })
}

dom만 그리느냐, 데이터도 모두 불러오느냐 (이미지 등)
